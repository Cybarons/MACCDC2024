#!/bin/bash

# Check if script is run as root
if [[ $EUID -ne 0 ]]; then
   echo "This script must be run as root" 
   exit 1
fi

# Function to reset user access
reset_user_access() {
    # Allow access for all users except root
    for user in $(getent passwd | awk -F ':' '{print $1}'); do
        if [ "$user" != "root" ]; then
            usermod -s /bin/bash "$user"
            echo "User '$user' access reset to normal"
        fi
    done
}

# Main function
main() {
    reset_user_access
}

# Run the main function
main
